48 bool SdlLibrary::drawCircle(std::pair<int, int> pos, int radius, Color color)
49 {
50  int diameter = (radius * 2);
51  int x = radius - 1;
52  int y = 0;
53  int tx = 1;
54  int ty = 1;
55  int error = (tx - diameter);
56 
57  while (x >= y) {
58  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) + x + 16, GRID_INT(pos.first) - y + 16);
59  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) + x + 16, GRID_INT(pos.first) + y + 16);
60  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) - x + 16, GRID_INT(pos.first) - y + 16);
61  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) - x + 16, GRID_INT(pos.first) + y + 16);
62  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) + y + 16, GRID_INT(pos.first) - x + 16);
63  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) + y + 16, GRID_INT(pos.first) + x + 16);
64  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) - y + 16, GRID_INT(pos.first) - x + 16);
65  SDL_RenderDrawPoint(_renderer, GRID_INT(pos.second) - y + 16, GRID_INT(pos.first) + x + 16);
66  if (error <= 0) {
67  y++;
68  error += ty;
69  ty += 2;
70  }
71  if (error > 0) {
72  x--;
73  tx += 2;
74  error += (tx - diameter);
75  }
76  SDL_SetRenderDrawColor(_renderer, color.R, color.G, color.B, color.A);
77  }
78 }
