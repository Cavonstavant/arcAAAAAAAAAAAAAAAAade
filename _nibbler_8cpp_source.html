156 void Nibbler::update(std::vector<std::shared_ptr<IEntity>> &entities, std::stack<Arcade::Evt> &events)

191 void Nibbler::moveSnake()
192 {
193  if (getClockTimeMS() * _speed > 200) {
194  resetClock();
195  IEntity::Direction dirPrev = _snake.front()->getDirection();
196  IEntity::Direction dirBuffer;

198  _lastTailPos = _snake.back()->getPos();
199  _lastDir = _snake.front()->getDirection();

202  it->setPos(getNextPos(it->getPos(), dirBuffer));
203  it->setDirection(dirPrev);

206  _gameState = GameState::STOPPED;

208  }
