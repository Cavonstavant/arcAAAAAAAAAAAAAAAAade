101 void Core::processEvents()
102 {
103  if (_event.empty())
104  return;
105  if (!_event.empty() && _event.top().evt_type == Arcade::Evt::EvtType::KEY)
106  return;
107  if (_event.top().evt_type == Arcade::Evt::WIN_CLOSE) {
108  if (_state == State::MAIN_MENU) {
109  _mainMenu.close(_entities);
110  _state = State::EXIT;
111  _event.pop();
112  return;
113  }
114  if (_state == State::GAME) {
115  _game->setState(IGame::GameState::STOPPED);
116  _state = State::MAIN_MENU;
117  _game = &_mainMenu;
118  _event.pop();
119  return;
120  }
121  }
122  if (_game->getState() == IGame::GameState::STOPPED) {
123  _state = State::MAIN_MENU;
124  _game = &_mainMenu;
125  _event.pop();
126  return;
127  }
128  if (_event.top().evt_type == Arcade::Evt::KEY) {
129  if (_event.top().key.key == 'r' || _event.top().key.key == 'R') {
130  try {
131  std::string currentGameName = _game->getLibraryName();
132  _libManager.closeLib(currentGameName);
133  _game = _libManager.openGame(currentGameName);
134  } catch (...) {
135  _event.pop();
136  return;
137  }
138  }
139  if (_event.top().key.key == 'm' || _event.top().key.key == 'M') {
140  _state = State::MAIN_MENU;
141  _game = &_mainMenu;
142  _event.pop();
143  return;
144  }
145  if (_event.top().key.key == 'p' || _event.top().key.key == 'P') {
146  _state = State::GAME;
147  _game->setState(IGame::GameState::PAUSED);
148  _event.pop();
149  return;
150  }
151  manageCoreKeyEvents(reinterpret_cast<std::string &>(_event.top().key.key));
152  _event.pop();
153  }
154 }
